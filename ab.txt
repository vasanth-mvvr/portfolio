// Avatar scroll effect
window.addEventListener("scroll", () => {
  const avatar = document.getElementById("timelineAvatar");
  const exp = document.querySelector(".experience");
  if (!avatar || !exp) return;

  const sectionTop = exp.offsetTop;
  const sectionHeight = exp.offsetHeight;
  const scrollY = window.scrollY;
  const windowH = window.innerHeight;
  const sectionBottom = sectionTop + sectionHeight;

  const start = sectionTop - windowH / 2;
  const end = sectionBottom - windowH / 2;

  if (scrollY >= start && scrollY <= end) {
    avatar.style.opacity = "1";
    avatar.style.position = "fixed";
    avatar.style.top = "50%";

    if (window.innerWidth <= 900) {
      avatar.style.left = "40px";
      avatar.style.transform = "translate(-50%, -50%)";
    } else {
      avatar.style.left = "50%";
      avatar.style.transform = "translate(-50%, -50%)";
    }
  } else {
    avatar.style.opacity = "0";
    avatar.style.position = "absolute";
    avatar.style.top = "0";

    if (window.innerWidth <= 900) {
      avatar.style.left = "40px";
      avatar.style.transform = "translateX(-50%)";
    } else {
      avatar.style.left = "50%";
      avatar.style.transform = "translate(-50%, 0)";
    }
  }
});
